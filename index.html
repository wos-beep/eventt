function copyTo(id) {
    const t = document.getElementById(id).innerText;
    navigator.clipboard.writeText(t).then(() => {
        const s = document.getElementById('toast');
        s.style.display = 'block';
        setTimeout(() => s.style.display = 'none', 2000);
    });
}
