function updateOutput() {
    const b = rawData.find(x => x.id === bSel.value);
    const oId = oSel.value;
    const shift = parseInt(document.getElementById('overlayShift').value) || 0;
    const rStart = parseInt(document.getElementById('rangeStart').value) || 1;
    
    let combined = {};
    let totalMax = b.days;
    let title = b.name;

    if (oId) {
        const o = rawData.find(x => x.id === oId);
        totalMax = Math.max(b.days, o.days + shift);
        title = `${b.name.substring(0,2)}+${o.name.substring(0,2)}`;
        new Set([...Object.keys(b.data), ...Object.keys(o.data)]).forEach(k => {
            let row = "";
            for (let d = 0; d < totalMax; d++) {
                const bV = (b.data[k] || "")[d] || "－";
                const oV = (o.data[k] || "")[d - shift] || "－";
                row += (bV !== "－" && oV !== "－") ? "◎" : (bV !== "－" ? bV : oV);
            }
            combined[k] = row;
        });
    } else { combined = b.data; totalMax = b.days; }

    let header = "日目|" + Array.from({length: totalMax - rStart + 1}, (_, i) => i + rStart).map(n => fullDigits[n] || n).join(' ') + "\n";
    let resultText = title + "\n" + header;
    
    let keys = Object.keys(combined);
    keys.forEach((k, idx) => {
        let line = `${padName(k)}|${combined[k].substring(rStart - 1, totalMax).split('').join(' ')}`;
        // 前半6項目は改行、後半はベタ打ち
        if (idx < 6) {
            resultText += line + "\n";
        } else {
            resultText += line; 
        }
    });

    document.getElementById('outputText').innerText = resultText.trim();
    document.getElementById('outputArea').style.display = 'block';
}

function step(id, val) {
    const el = document.getElementById(id);
    let newVal = parseInt(el.value) + val;
    if (id === 'rangeStart') newVal = Math.min(8, Math.max(1, newVal));
    if (id === 'overlayShift') newVal = Math.max(0, newVal);
    el.value = newVal;
    updateOutput();
}

function copyToClipboard() {
    navigator.clipboard.writeText(document.getElementById('outputText').innerText).then(() => {
        const s = document.getElementById('toast');
        s.style.display = 'block';
        setTimeout(() => s.style.display = 'none', 2000);
    });
}
